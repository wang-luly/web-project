错误处理
document.getElementById('btnError').addEventListener('click',async()=>{
  resetPage();
  loading.style.display='block';
  try{
    const response=await fetch("https://jsonplaceholder.typicode.com/invalid-url-404");
    if(!response.ok){
      if(response.status==404){
        throw new Error('404错误，请求的内容不存在')；
      }
      else if(response.status==500){
        throw new Error('500错误：服务器错误')
      }
      else{
        throw new Error(`http错误！状态码：${response.status}`);
      }
    }
    const data=await response.json();
    console.log("错误处理data值：",data);
  }catch(error){
    console.error('错误处理，捕获到异常',error);
    errorMsg.textContent=`错误：${error.message}`;
  }finally{
    loading.style.display='none';
  }
})


// 3. 数据渲染：获取列表数据并动态展示到 HTML 页面
document.getElementById('btnRender').addEventListener('click',async()=>{
  resetPage();
  loading.style.display='block';
  try{
    const response=await fetch("https://jsonplaceholder.typicode.com/posts?_limit=1");
    if(!response.ok){
      throw new Error(`http错误，状态码${response.status}`);
    }
    //post is an array
    const posts=await response.json();
    posts.forEach((post)=>{
      const li=document.createElement('li');
      li.innerHTML=`
        <strong>标题:${post.title}<strong>
        <p>内容:${post.body}<p>
      `;
      dataList.appendChild(li);
    })
    cosole.log('帖子列表',posts);
  }catch(error){
    console.error('数据渲染出错了',error);
    errorMsg.textContent=`加载帖子失败:${error.message}`;
  }finally{
    loading.style.display='none';
  }
});
